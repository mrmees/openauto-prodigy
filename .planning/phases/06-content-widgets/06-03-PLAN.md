    **3. Wire in main.cpp:**

    After the AA plugin initialization and before widget registration:
    ```cpp
    #include "core/aa/NavigationDataBridge.hpp"
    #include "core/aa/MediaDataBridge.hpp"
    #include "core/aa/ManeuverIconProvider.hpp"

    // Create bridges
    auto* navBridge = new oap::aa::NavigationDataBridge(&app);
    auto* mediaBridge = new oap::aa::MediaDataBridge(&app);
    auto* maneuverIconProvider = new oap::aa::ManeuverIconProvider();

    // Wire nav bridge to orchestrator's nav handler
    navBridge->connectToHandler(aaOrchestrator->navigationHandler());
    navBridge->setManeuverIconProvider(maneuverIconProvider);

    // Wire media bridge
    mediaBridge->connectToAAOrchestrator(aaOrchestrator);
    mediaBridge->connectToBtAudio(btAudioPlugin);  // btAudioPlugin may be nullptr on VM

    // Register image provider (engine takes ownership)
    engine.addImageProvider("navicon", maneuverIconProvider);

    // Expose as root context properties
    engine.rootContext()->setContextProperty("NavigationBridge", navBridge);
    engine.rootContext()->setContextProperty("MediaBridge", mediaBridge);
    ```
