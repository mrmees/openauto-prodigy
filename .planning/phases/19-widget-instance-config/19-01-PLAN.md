    **`validateConfig` implementation** (per CONTEXT.md requirement: host validates against schema before persisting):
    ```cpp
    QVariantMap WidgetGridModel::validateConfig(const QString& widgetId, const QVariantMap& raw) const {
        if (!registry_) return {};
        auto desc = registry_->descriptor(widgetId);
        if (!desc || desc->configSchema.isEmpty()) return {};

        // Build schema key lookup
        QHash<QString, const ConfigSchemaField*> schemaMap;
        for (const auto& field : desc->configSchema)
            schemaMap[field.key] = &field;

        QVariantMap validated;
        for (auto it = raw.constBegin(); it != raw.constEnd(); ++it) {
            auto* field = schemaMap.value(it.key(), nullptr);
            if (!field) continue;  // reject unknown keys

            switch (field->type) {
            case ConfigFieldType::Bool: {
                // Coerce to bool
                QVariant v = it.value();
                if (v.typeId() == QMetaType::QString) {
                    validated[it.key()] = (v.toString().toLower() == "true");
                } else {
                    validated[it.key()] = v.toBool();
                }
                break;
            }
            case ConfigFieldType::IntRange: {
                int val = it.value().toInt();
                val = qBound(field->rangeMin, val, field->rangeMax);
                validated[it.key()] = val;
                break;
            }
            case ConfigFieldType::Enum: {
                // Accept only if value is in the values list
                if (field->values.contains(it.value())) {
                    validated[it.key()] = it.value();
                }
                // else: silently drop invalid enum value (default will apply)
                break;
            }
            }
        }
        return validated;
    }
    ```

    **`setWidgetConfig` implementation:** Find placement by instanceId. Call `validateConfig(placement.widgetId, config)` to get validated overrides. Set `livePlacements_[idx].config = validated`. Emit `dataChanged` for that index with `{ConfigRole}`. Call `promoteToBase()`. Emit `placementsChanged()`. Compute effective config by merging defaults + validated overrides and emit `widgetConfigChanged(instanceId, effectiveConfig)`.

    ```cpp
    void WidgetGridModel::setWidgetConfig(const QString& instanceId, const QVariantMap& config) {
        int idx = findPlacement(instanceId);
        if (idx < 0) return;

        auto& p = livePlacements_[idx];
        QVariantMap validated = validateConfig(p.widgetId, config);
        p.config = validated;

        QModelIndex mi = index(idx);
        emit dataChanged(mi, mi, {ConfigRole});
        promoteToBase();
        emit placementsChanged();

        // Compute effective config (defaults + validated overrides) for live notification
        QVariantMap effective;
        if (registry_) {
            auto desc = registry_->descriptor(p.widgetId);
            if (desc) effective = desc->defaultConfig;
        }
        for (auto it = validated.constBegin(); it != validated.constEnd(); ++it)
            effective[it.key()] = it.value();

        emit widgetConfigChanged(instanceId, effective);
    }
    ```

    **`widgetConfig` implementation:** Find placement, return `livePlacements_[idx].config` (raw overrides only) or empty map if not found.

    **`effectiveWidgetConfig` implementation:** Find placement, merge descriptor's `defaultConfig` with `livePlacements_[idx].config` (instance overrides win), return the merged map. Returns empty map if not found.

    ```cpp
    QVariantMap WidgetGridModel::effectiveWidgetConfig(const QString& instanceId) const {
        int idx = findPlacement(instanceId);
        if (idx < 0) return {};

        const auto& p = livePlacements_[idx];
        QVariantMap result;
        if (registry_) {
            auto desc = registry_->descriptor(p.widgetId);
            if (desc) result = desc->defaultConfig;
        }
        for (auto it = p.config.constBegin(); it != p.config.constEnd(); ++it)
            result[it.key()] = it.value();
        return result;
    }
    ```

    In `WidgetContextFactory.cpp`, update `createContext`:
    ```cpp
    QObject* WidgetContextFactory::createContext(const QString& instanceId, QObject* parent) {
        if (!gridModel_) return nullptr;

        const auto& placements = gridModel_->placements();
        for (const auto& p : placements) {
            if (p.instanceId == instanceId) {
                QVariantMap defaultCfg;
                if (auto* reg = gridModel_->registry()) {
                    auto desc = reg->descriptor(p.widgetId);
                    if (desc) defaultCfg = desc->defaultConfig;
                }
                auto* ctx = new WidgetInstanceContext(p, cellSide_, cellSide_, hostContext_,
                                                      defaultCfg, p.config, parent);

                // Track this context for live config updates
                activeContexts_[instanceId] = ctx;

                // Clean up tracking when context is destroyed (delegate removed/recycled)
                connect(ctx, &QObject::destroyed, this, [this, instanceId]() {
                    activeContexts_.remove(instanceId);
                });

                return ctx;
            }
        }
        return nullptr;
    }
    ```

    In the `WidgetContextFactory` constructor, connect to `WidgetGridModel::widgetConfigChanged`:
    ```cpp
    WidgetContextFactory::WidgetContextFactory(WidgetGridModel* gridModel,
                                               IHostContext* hostContext,
                                               QObject* parent)
        : QObject(parent)
        , gridModel_(gridModel)
        , hostContext_(hostContext)
    {
        if (gridModel_) {
            connect(gridModel_, &WidgetGridModel::widgetConfigChanged,
                    this, [this](const QString& instanceId, const QVariantMap& effectiveConfig) {
                auto* ctx = activeContexts_.value(instanceId, nullptr);
                if (ctx) {
                    // Extract just the instance overrides (effectiveConfig - defaults)
                    // But simpler: WidgetGridModel already stores validated overrides.
                    // We need the raw overrides to call setInstanceConfig, which re-merges.
                    // The effectiveConfig param IS the merged result, but setInstanceConfig
                    // expects overrides only. Instead, read overrides from the model.
                    ctx->setInstanceConfig(gridModel_->widgetConfig(instanceId));
                }
            });
        }
    ```
